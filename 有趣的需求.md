# 一、字符串检验
## 1.有一批字符串输入，每个字符串中可能会包含很多括号，过滤掉不符合规则的，
> 要求：
>   1. 括号不能嵌套
>   2. 括号中不能为空
>   3. 不能出现单个括号
>

```java
private boolean verification(String str) {
    int a = getNum(str, "(\\()");
    int b = getNum(str, "(\\))");
    if (!(a != 0 && a == b)) {
        return false;
    }
    Pattern compile = Pattern.compile("(\\(.*?\\))");
    Matcher matcher = compile.matcher(str);
    while (matcher.find()) {
        String content = matcher.group(1);
        content = content.substring(1, content.length() - 1);
        if (content.length() <= 0 || getNum(content, "(\\()") > 0 || getNum(content, "(\\))") > 0) {
            return false;
        }
    }
    return true;
}

private int getNum(String str, String pattern) {
    int temp = 0;
    Pattern compile = Pattern.compile(pattern);
    Matcher matcher = compile.matcher(str);
    while (matcher.find()) {
        String x = matcher.group(1);
        if (x != null && x.length() > 0) {
            temp++;
        }
    }
    return temp;
}
```

```php
function check($str) {
    $left_count = 0;
    $right_count = 0;
    for ($i = 0; $i < mb_strlen($str); $i++) {
        // 获取第 i 个字符
        $char = mb_substr($str, $i, 1);
        if ($char == '(') {
            // ++之前判断，此时(和)的数量应该相等，否则++后(会多一个，解决嵌套问题
            if ($left_count != $right_count) {
                return false;
            }
            $left_count++;
        } else if ($char == ')') {
            $right_count++;
            // ++后，此时(和)的数量应该相等，否则)多一个
            // 括号内不能为空，所以)前面一位不能是(
            if ($left_count != $right_count || $i == 0 || mb_substr($str, $i - 1, 1) == '(') {
                return false;
            }
        }
    }
    // 如果数量不相等，则GG
    if ($left_count != $right_count) {
        return false;
    }
    return true;
}
```
